## Install

```bash
npm install --save react-ransom-letters
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-ransom-letters'
import 'react-ransom-letters/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
